PROGRAM DIINP
   DEFINE VAR VAR02 VAR03 AUX INT FRA
   PRINT "TYPE VAR01"
   READ VAR01
   PRINT "TYPE VAR02"
   READ VAR02
   
   SET AUX = VAR01 * 100
   SET VAR03 = AUX / VAR02
   SET INT = VAR03 / 100
   SET FRA = VAR03 MOD 100
   PRINTLN "RESULT =" INT "," FRA
     
END
